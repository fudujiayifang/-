function openDrawer(drawer, overlay, btn) {
  drawer.classList.add('open');
  overlay.classList.add('active');
  btn.classList.add('active');
}
function closeDrawer(drawer, overlay, btn) {
  drawer.classList.remove('open');
  overlay.classList.remove('active');
  btn.classList.remove('active');
}
function closeAllDrawers() {
  closeDrawer(drawerLeft, overlayLeft, btnLeft);
  closeDrawer(drawerRight, overlayRight, btnRight);
}

btnLeft.addEventListener('click', () => {
  if (drawerLeft.classList.contains('open')) {
    closeDrawer(drawerLeft, overlayLeft, btnLeft);
  } else {
    closeDrawer(drawerRight, overlayRight, btnRight);
    openDrawer(drawerLeft, overlayLeft, btnLeft);
  }
});
btnRight.addEventListener('click', () => {
  if (drawerRight.classList.contains('open')) {
    closeDrawer(drawerRight, overlayRight, btnRight);
  } else {
    closeDrawer(drawerLeft, overlayLeft, btnLeft);
    openDrawer(drawerRight, overlayRight, btnRight);
  }
});
overlayLeft.addEventListener('click',  () => closeDrawer(drawerLeft, overlayLeft, btnLeft));
overlayRight.addEventListener('click', () => closeDrawer(drawerRight, overlayRight, btnRight));

function openModal(key, label) {
  const root = document.getElementById('modal-root');
  const config = modalConfig[key];
  if (!config) {
    showToast(`「${label}」功能将在后续版本中完善`, 'info');
    return;
  }

  if (config.bag) {
    // Render as bag grid — only real items, pad last row with phantom cells
    const real = config.items.filter(Boolean);
    const pad = (5 - (real.length % 5)) % 5;
    const displayItems = [...real, ...Array(pad).fill({phantom:true})];

    const gridHTML = displayItems.map((item, i) => {
      if (item.phantom) return '<div class="bag-cell phantom"></div>';
      return `<div class="bag-cell" title="${item.name}" data-idx="${i}">
        ${iconSVGs[item.icon] || iconSVGs.star}
        ${item.count > 1 ? `<span class="cell-count">${item.count}</span>` : ''}
      </div>`;
    }).join('');

    const modalHTML = `
      <div class="modal-overlay" id="current-modal-overlay" onclick="event.target===this&&this.parentElement&&document.getElementById('modal-root').innerHTML=''">
        <div class="modal" role="dialog" aria-labelledby="modal-title" style="min-width:420px;max-width:440px;">
          <button class="modal-close" id="modal-close-btn" aria-label="关闭">&times;</button>
          <div class="modal-title" id="modal-title">${config.title}</div>
          <div class="modal-subtitle">${config.subtitle}</div>
          <div class="bag-grid">${gridHTML}</div>
        </div>
      </div>
    `;

    root.innerHTML = modalHTML;

    const overlay = document.getElementById('current-modal-overlay');
    const closeBtn = document.getElementById('modal-close-btn');
    function closeModal() { root.innerHTML = ''; }
    closeBtn.addEventListener('click', closeModal);
    overlay.addEventListener('click', (e) => { if (e.target === overlay) closeModal(); });
    document.addEventListener('keydown', function escH(e) {
      if (e.key === 'Escape') { closeModal(); document.removeEventListener('keydown', escH); }
    });

    // Click item to show toast
    if (!config._bound) {
      root.addEventListener('click', (e) => {
        const cell = e.target.closest('.bag-cell:not(.phantom)');
        if (cell) {
          const idx = parseInt(cell.dataset.idx);
          const realItems = config.items.filter(Boolean);
          const item = realItems[idx];
          if (item) showToast(`${item.name} — ${item.count > 1 ? 'x'+item.count : '点击查看详情'}`, 'info');

      });
      config._bound = true;

          doBtn.disabled = false;
          doBtn.className = 'checkin-btn';
          doBtn.textContent = '呼唤零七签到';
        }
      }
    }

    initCalendarModal();
    return;
  }

  const itemsHTML = config.items.map(item => `
    <div class="modal-grid-item">
      ${iconSVGs[item.icon] || iconSVGs.star}
      <div class="item-name">${item.name}</div>
      <div class="item-meta">${item.meta}${item.tag ? ' <span class="tag tag-'+item.tag+'">'+item.tag+'</span>' : ''}</div>
    </div>
  `).join('');

  const modalHTML = `
    <div class="modal-overlay" id="current-modal-overlay">
      <div class="modal" role="dialog" aria-labelledby="modal-title">
        <button class="modal-close" id="modal-close-btn" aria-label="关闭">&times;</button>
        <div class="modal-title" id="modal-title">${config.title}</div>
        <div class="modal-subtitle">${config.subtitle}</div>
        <div class="modal-grid">${itemsHTML}</div>
      </div>
    </div>
  `;

  root.innerHTML = modalHTML;

  const overlay = document.getElementById('current-modal-overlay');
  const closeBtn = document.getElementById('modal-close-btn');

  function closeModal() {
    overlay.style.opacity = '0';
    overlay.querySelector('.modal').style.transform = 'scale(0.92) translateY(20px)';
    overlay.querySelector('.modal').style.opacity = '0';
    overlay.querySelector('.modal').style.transition = 'all 0.2s ease-in';
    setTimeout(() => { root.innerHTML = ''; }, 200);
  }

  closeBtn.addEventListener('click', closeModal);
  overlay.addEventListener('click', (e) => { if (e.target === overlay) closeModal(); });
  document.addEventListener('keydown', function escHandler(e) {
    if (e.key === 'Escape') { closeModal(); document.removeEventListener('keydown', escHandler); }
  });
}

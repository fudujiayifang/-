    function makeBar(val, max, colorClass) {
      var pct = Math.min(100, Math.max(0, (val / max) * 100));
      return '<div class="bar-wrap">' +
        '<div class="bar-track"><div class="bar-fill ' + colorClass + '" style="width:' + pct + '%"></div></div>' +
        '<span class="bar-num">' + val + '</span></div>';
    }

    function renderRecord(obj, renderFn) {
      var entries = Object.entries(obj || {});
      if (entries.length === 0) return '<div class="empty">暂无</div>';
      return entries.map(renderFn).join('');
    }

    function populateCharacterData() {

    function updateSliderPosition() {
      if (totalSlides === 0) return;
      if (currentSlide < 0) currentSlide = 0;
      if (currentSlide >= totalSlides) currentSlide = totalSlides - 1;
      $('#slider-track').css('transform', 'translateX(-' + (currentSlide * 100) + '%)');
      $('.slider-dot').removeClass('active');
      $('.slider-dot[data-idx="' + currentSlide + '"]').addClass('active');

      var all = getAllVariables();
      var v = _.get(all, 'stat_data', {});
      var names = Object.keys(_.get(v, '攻略目标', {}));
      $('#slider-name').text(names[currentSlide] || '无目标');
    }

    function switchTab(tabId) {
      $('.tab-btn').removeClass('active');
      $('.tab-btn[data-tab="' + tabId + '"]').addClass('active');
      $('.page').removeClass('active');
      $('#page-' + tabId).addClass('active');
    }

    async function init() {
      await waitGlobalInitialized('Mvu');
      populateCharacterData();

      eventOn(Mvu.events.VARIABLE_UPDATE_ENDED, function() {
        populateCharacterData();
      });

      $('.tab-btn').on('click', function() {
        switchTab($(this).data('tab'));
      });

      $(document).on('click', '.slider-dot', function() {
        currentSlide = parseInt($(this).data('idx'));
        updateSliderPosition();
      });
      $('#slider-prev').on('click', function() {
        if (currentSlide > 0) { currentSlide--; updateSliderPosition(); }
      });
      $('#slider-next').on('click', function() {
        if (currentSlide < totalSlides - 1) { currentSlide++; updateSliderPosition(); }
      });

      var sliderEl = document.getElementById('slider-wrap');
      if (sliderEl) {
        sliderEl.addEventListener('touchstart', function(e) {
          touchStartX = e.changedTouches[0].screenX;
        }, { passive: true });
        sliderEl.addEventListener('touchend', function(e) {
          touchEndX = e.changedTouches[0].screenX;
          var diff = touchStartX - touchEndX;
          if (Math.abs(diff) > 50) {
            if (diff > 0 && currentSlide < totalSlides - 1) { currentSlide++; }
            else if (diff < 0 && currentSlide > 0) { currentSlide--; }
            updateSliderPosition();
          }
        }, { passive: true });
      }
    }

    $(errorCatched(init));
